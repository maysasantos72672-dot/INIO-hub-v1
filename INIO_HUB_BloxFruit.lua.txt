-- ===================================
-- Funções de Funcionalidades
-- ===================================

local function UpdateSpeed()
    if Settings.SpeedEnabled and Character and Humanoid then
        Humanoid.WalkSpeed = Settings.SpeedValue
    else
        Humanoid.WalkSpeed = 16
    end
end

local function UpdateJumpPower()
    if Settings.JumpPowerEnabled and Character and Humanoid then
        Humanoid.JumpPower = Settings.JumpPowerValue
    else
        Humanoid.JumpPower = 50
    end
end

local function EnableNoClip()
    if Settings.NoClipEnabled and Character then
        for _, part in pairs(Character:GetDescendants()) do
            if part:IsA("BasePart") then
                part.CanCollide = false
            end
        end
    end
end

-- ===================================
-- Lógica de Auto Farm Avançada
-- ===================================

local FarmTargets = {}
local LastAttackTime = 0

function AutoFarmLogic()
    if not Settings.AutoFarmEnabled or not RootPart then return end

    FarmTargets = {}

    -- Buscar Inimigos
    if Settings.FarmEnemies then
        local Workspace = game:GetService("Workspace")
        
        -- Procurar em pastas de inimigos comuns no Blox Fruit
        local EnemyFolders = {
            Workspace:FindFirstChild("Enemies"),
            Workspace:FindFirstChild("NPCs"),
            Workspace:FindFirstChild("Mobs"),
            Workspace:FindFirstChild("Bosses"),
            Workspace:FindFirstChild("Pirates"),
            Workspace:FindFirstChild("Marines"),
        }

        for _, folder in ipairs(EnemyFolders) do
            if folder then
                for _, enemy in pairs(folder:GetChildren()) do
                    if enemy:FindFirstChild("Humanoid") and enemy:FindFirstChild("HumanoidRootPart") then
                        local distance = (RootPart.Position - enemy.HumanoidRootPart.Position).Magnitude
                        if distance <= Settings.FarmDistance then
                            table.insert(FarmTargets, {
                                Object = enemy,
                                Distance = distance,
                                Type = "Enemy"
                            })
                        end
                    end
                end
            end
        end
    end

    -- Buscar Jogadores (PvP)
    if Settings.FarmPlayers then
        for _, player in pairs(game:GetService("Players"):GetPlayers()) do
            if player ~= LocalPlayer and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
                local distance = (RootPart.Position - player.Character.HumanoidRootPart.Position).Magnitude
                if distance <= Settings.FarmDistance then
                    table.insert(FarmTargets, {
                        Object = player.Character,
                        Distance = distance,
                        Type = "Player"
                    })
                end
            end
        end
    end

    -- Coletar Frutas
    if Settings.FarmFruits then
        local Workspace = game:GetService("Workspace")
        local fruitFolders = {
            Workspace:FindFirstChild("Fruits"),
            Workspace:FindFirstChild("DroppedFruits"),
            Workspace:FindFirstChild("Items"),
        }
        
        for _, folder in ipairs(fruitFolders) do
            if folder then
                for _, fruit in pairs(folder:GetChildren()) do
                    if fruit:FindFirstChild("HumanoidRootPart") or fruit:IsA("BasePart") then
                        local pos = fruit:FindFirstChild("HumanoidRootPart") and fruit.HumanoidRootPart.Position or fruit.Position
                        local distance = (RootPart.Position - pos).Magnitude
                        if distance <= Settings.FarmDistance then
                            RootPart.CFrame = CFrame.new(pos + Vector3.new(0, 3, 0))
                        end
                    end
                end
            end
        end
    end

    -- Coletar Dinheiro
    if Settings.FarmCash then
        local Workspace = game:GetService("Workspace")
        local cashFolders = {
            Workspace:FindFirstChild("Cash"),
            Workspace:FindFirstChild("Money"),
            Workspace:FindFirstChild("Drops"),
        }
        
        for _, folder in ipairs(cashFolders) do
            if folder then
                for _, cash in pairs(folder:GetChildren()) do
                    local distance = (RootPart.Position - cash.Position).Magnitude
                    if distance <= Settings.FarmDistance then
                        RootPart.CFrame = CFrame.new(cash.Position + Vector3.new(0, 3, 0))
                    end
                end
            end
        end
    end

    -- Ordenar por distância
    table.sort(FarmTargets, function(a, b)
        return a.Distance < b.Distance
    end)

    -- Mover e Atacar alvo mais próximo
    if #FarmTargets > 0 then
        local target = FarmTargets[1]
        local targetPos = target.Object:FindFirstChild("HumanoidRootPart") and target.Object.HumanoidRootPart.Position or target.Object.Position

        -- Mover em direção ao alvo
        local direction = (targetPos - RootPart.Position).Unit
        RootPart.CFrame = RootPart.CFrame + direction * 2

        -- Auto Ataque
        if Settings.AutoAttack then
            local currentTime = tick()
            if currentTime - LastAttackTime > 0.5 then
                -- Simular ataque básico
                if target.Object:FindFirstChild("Humanoid") then
                    target.Object.Humanoid:TakeDamage(10)
                end
                LastAttackTime = currentTime
            end
        end
    end
end

function TeleportToClosestEnemy()
    local Workspace = game:GetService("Workspace")
    local closestEnemy = nil
    local closestDistance = math.huge

    local EnemyFolders = {
        Workspace:FindFirstChild("Enemies"),
        Workspace:FindFirstChild("NPCs"),
        Workspace:FindFirstChild("Mobs"),
    }

    for _, folder in ipairs(EnemyFolders) do
        if folder then
            for _, enemy in pairs(folder:GetChildren()) do
                if enemy:FindFirstChild("HumanoidRootPart") then
                    local distance = (RootPart.Position - enemy.HumanoidRootPart.Position).Magnitude
                    if distance < closestDistance then
                        closestEnemy = enemy
                        closestDistance = distance
                    end
                end
            end
        end
    end

    if closestEnemy then
        RootPart.CFrame = closestEnemy.HumanoidRootPart.CFrame + Vector3.new(0, 5, 0)
    end
end

function ResetPosition()
    RootPart.CFrame = CFrame.new(0, 50, 0)
    Humanoid:ChangeHealth(Humanoid.MaxHealth)
end
